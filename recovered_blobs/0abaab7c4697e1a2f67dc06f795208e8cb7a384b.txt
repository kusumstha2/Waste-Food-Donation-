  <script>
    // Function to view donation details
    function viewDetails(id) {
      document.querySelector(`#card-${id} .view-details-btn`).style.display = 'none';
      document.querySelector(`#card-${id} .edit-btn`).style.display = 'none';
      document.querySelector(`#card-${id} .delete-btn`).style.display = 'none';
      
      // Show the details section
      document.querySelector(`#details-${id}`).style.display = 'block';
    }

    // Function to go back to the overview
    function backToOverview(id) {
      document.querySelector(`#card-${id} .view-details-btn`).style.display = 'inline-block';
      document.querySelector(`#card-${id} .edit-btn`).style.display = 'inline-block';
      document.querySelector(`#card-${id} .delete-btn`).style.display = 'inline-block';
      
      // Hide the details section
      document.querySelector(`#details-${id}`).style.display = 'none';
    }

    // Example functions for edit and delete (to be implemented)
    function editDonation(id) {
      // Use a data attribute to get the base URL generated by Django
      const baseEditUrl = document.getElementById('/editdonation/').dataset.url;
      window.location.href = `${baseEditUrl}${id}/`;
    }
    
    function deleteDonation(id) {
      if (confirm("Are you sure you want to delete this donation?")) {
        window.location.href = `delete_donation/${id}/`;
      }
    }
  </script>
